<section>
  <p class="section-title reveal">🎯 <span>Key Takeaways</span></p>
  <p class="section-sub reveal">The three things this project proves, and the three things it still needs.</p>

  <div class="grid-2 reveal">
    <div class="card green">
      <h3 style="color:var(--accent3);margin-bottom:16px;">✅ What We've Proven</h3>
      <div class="highlight-box green">
        <strong>Lag features dominate prediction</strong>
        <p>Same-hour yesterday (lag_24h) and same-hour last week (lag_168h) are the single most predictive signals — justifying the LSTM's 168h lookback window.</p>
      </div>
      <div class="highlight-box green">
        <strong>Multi-seasonality is real and exploitable</strong>
        <p>Daily, weekly, and annual cycles all explain significant variance. Cyclic sin/cos encoding is essential for XGBoost to learn these boundaries correctly.</p>
      </div>
      <div class="highlight-box green">
        <strong>XGBoost achieves near-sota with features alone</strong>
        <p>Starting RMSE of 1,472 MWh drops to ~86 MWh purely via feature engineering — demonstrating that data quality beats model complexity.</p>
      </div>
    </div>
    <div class="card red">
      <h3 style="color:var(--accent5);margin-bottom:16px;">⚠️ Critical Gaps</h3>
      <div class="highlight-box" style="border-color:var(--accent5);background:rgba(239,68,68,.05);">
        <strong>No holiday awareness</strong>
        <p>Christmas, New Year, and national holidays cause 15–25% demand drops that no current feature explains. This is the #1 missing feature.</p>
      </div>
      <div class="highlight-box" style="border-color:var(--accent5);background:rgba(239,68,68,.05);">
        <strong>No exogenous variables</strong>
        <p>The model only sees past consumption — not weather, economic activity, or scheduled events. Real grid forecasters always include weather as a primary input.</p>
      </div>
      <div class="highlight-box" style="border-color:var(--accent5);background:rgba(239,68,68,.05);">
        <strong>No uncertainty quantification</strong>
        <p>Point predictions without confidence intervals are risky for grid operations. Operators need to know not just the forecast, but how uncertain it is.</p>
      </div>
    </div>

  const heatmap = document.getElementById('heatmap');

  // Simulated relative consumption intensity (0-1) for each day × hour
  // Based on EDA findings: bimodal daily, weekday > weekend, morning + evening peaks
  function intensity(day, hour) {
    const isWeekend = day >= 5;
    const weekendFactor = isWeekend ? 0.82 : 1.0;
    // Morning ramp 8-10, evening peak 17-20, overnight valley 3-5
    let h;
    if (hour >= 0  && hour <= 2)  h = 0.35;
    else if (hour >= 3 && hour <= 5)  h = 0.28;
    else if (hour >= 6 && hour <= 7)  h = 0.52;
    else if (hour >= 8 && hour <= 10) h = 0.80;
    else if (hour >= 11 && hour <= 13) h = 0.72;
    else if (hour >= 14 && hour <= 16) h = 0.68;
    else if (hour >= 17 && hour <= 20) h = 0.95;
    else if (hour >= 21 && hour <= 22) h = 0.60;
    else h = 0.42;
    return Math.min(1, h * weekendFactor + (Math.random() * 0.04 - 0.02));
  }

  function intensityToColor(v) {
    // low = dark blue, high = bright cyan
    const r = Math.round(14 + v * (0  - 14));
    const g = Math.round(58 + v * (212 - 58));
    const b = Math.round(100 + v * (255 - 100));
    return `rgb(${r},${g},${b})`;
  }

  // Header row
  heatmap.appendChild(Object.assign(document.createElement('div'), { className:'heatmap-header' }));
  for (let h = 0; h < 24; h++) {
    const d = document.createElement('div');
    d.className = 'heatmap-header';
    d.textContent = h;
    heatmap.appendChild(d);
  }

  // Data rows
  days.forEach((day, di) => {
    const lbl = document.createElement('div');
    lbl.className = 'heatmap-row-label';
    lbl.textContent = day;
    heatmap.appendChild(lbl);
    for (let h = 0; h < 24; h++) {
      const v = intensity(di, h);
      const cell = document.createElement('div');
      cell.className = 'heatmap-cell';
      cell.style.background = intensityToColor(v);
      cell.style.opacity = 0.7 + v * 0.3;
      heatmap.appendChild(cell);
    }
